00020         {

00026            float XSquared = X * X;
00027            float YSquared = Y * Y;

00029            TEuler.setX(atan2f(2.0f * (Y * Z + X * W), -XSquared - YSquared + ZSquared + WSquared));

00031            TEuler.setZ(atan2f(2.0f * (X * Y + Z * W), XSquared - YSquared - ZSquared + WSquared));
00032            TEuler *= irr::core::RADTODEG;
00033         }
